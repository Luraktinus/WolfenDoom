// WALL DECS //
ACTOR Wall_CampMalePrisoner : WallDecoBase
{
	//$Category Wall Decorations (Wolf3D) /Concentration Camp
	//$Title Camp Prisoner Hanging (male, random)
	//$Color 3
	Radius 64
	Height 64
	Scale 0.65
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_Jump(256,1,2,3,4,5,6,7,8,10)
		CHGM A -1
		CHGM E -1
		CHGM F -1
		CHGM G -1
		CHGM H -1
		CHGM I -1
		CHGM J -1
		CHGM K -1
		Stop
	Moving:
		"####" B 1 A_SetTics(Random(80,160))
		"####" C 8
		"####" D 8
		"####" C 8
		Loop
	}
}

ACTOR Wall_CivilianMale4 : Wall_CampMalePrisoner
{
	//$Category Wall Decorations (Wolf3D)
	//$Title Civilian Prisoner Hanging (smoking, shootable)
	Health 25
	Mass 0x7ffffff
	+DONTFALL
	+SHOOTABLE
	States
	{
	Spawn:
		CIV4 A 1 A_JumpIfHealthLower(40,"Bleeding")
		Loop
	Bleeding:
		CIV4 B 1 A_SetTics(Random(80,160))
		"####" C 8
		"####" D 8
		"####" B 8
		Loop
	Death:
		TNT1 A 0 A_Jump(256,1,2)
		CIV4 E -1 A_UnSetShootable
		CIV4 F -1 A_UnSetShootable
		Stop
	}
}
